                </div>
                <div>
                    <label class="block text-sm font-semibold mb-1 text-slate-600">Платено в €</label>

                </div>

    <script>
        const RATE = 1.95583;
        let changeCurrency = 'EUR'; // По подразбиране [cite: 24]

        // Генериране на фон [cite: 8, 9]
        const pattern = document.getElementById('pattern');
        for (let i = 0; i < 40; i++) {
            const span = document.createElement('span');
            span.className = 'flex items-center justify-center text-2xl float-icon';
            span.style.animationDelay = `${Math.random() * 5}s`;
            span.innerText = i % 2 === 0 ? '€' : 'лв';
            pattern.appendChild(span);
        }

        // Логика за изчисляване [cite: 28]
        function calculate() {
            const total = parseFloat(document.getElementById('totalAmount').value) || 0;
            const totalCurr = document.getElementById('totalCurrency').value;
            const paidBGN = parseFloat(document.getElementById('paidBGN').value) || 0;
            const paidEUR = parseFloat(document.getElementById('paidEUR').value) || 0;

            // Всичко към лева за базова сметка
            const totalInBGN = totalCurr === 'EUR' ? total * RATE : total;
            const totalPaidInBGN = paidBGN + (paidEUR * RATE);
            const changeInBGN = totalPaidInBGN - totalInBGN;

            if (total > 0 && changeInBGN < -0.01) {
                document.getElementById('errorBox').classList.remove('hidden');
                document.getElementById('resultBox').classList.add('hidden');
            } else if (total > 0 && changeInBGN >= 0) {
                document.getElementById('errorBox').classList.add('hidden');
                document.getElementById('resultBox').classList.remove('hidden');
                updateDisplay(changeInBGN);
            } else {
                document.getElementById('errorBox').classList.add('hidden');
                document.getElementById('resultBox').classList.add('hidden');
            }
        }

        function updateDisplay(bgn) {
            const mainEl = document.getElementById('mainChange');
            const altEl = document.getElementById('altChange');
            
            if (changeCurrency === 'BGN') {
                mainEl.innerText = `${bgn.toFixed(2)} лв`;
                altEl.innerText = `Еквивалент: ${(bgn / RATE).toFixed(2)} €`;
            } else {
                const eur = bgn / RATE;
                mainEl.innerText = `${eur.toFixed(2)} €`;
                altEl.innerText = `Еквивалент: ${bgn.toFixed(2)} лв`;
            }
        }

        // Слушатели [cite: 30]
        document.querySelectorAll('input, select').forEach(el => el.addEventListener('input', calculate));
        
        document.getElementById('btnBGN').onclick = () => {
            changeCurrency = 'BGN';
            document.getElementById('btnBGN').classList.add('bg-white', 'text-blue-700');
            document.getElementById('btnEUR').classList.remove('bg-white', 'text-blue-700');
            calculate();
        };

        document.getElementById('btnEUR').onclick = () => {
            changeCurrency = 'EUR';
            document.getElementById('btnEUR').classList.add('bg-white', 'text-blue-700');
            document.getElementById('btnBGN').classList.remove('bg-white', 'text-blue-700');
            calculate();
        };

        // PWA Регистриране на Service Worker [cite: 20, 21, 23]
        const swCode = `self.addEventListener('fetch', function(e) { e.respondWith(fetch(e.request)); });`;
        const blob = new Blob([swCode], {type: 'text/javascript'});
        const swUrl = URL.createObjectURL(blob);
        if ('serviceWorker' in navigator) {
            navigator.serviceWorker.register(swUrl);
        }

        // Web Manifest [cite: 20]
        const manifest = {
            name: "DualPay Calculator",
            short_name: "DualPay",
            start_url: ".",
            display: "standalone",
            background_color: "#f8fafc",
            theme_color: "#1d4ed8",
            icons: [{ src: "https://via.placeholder.com/192/1d4ed8/ffffff?text=€/лв", sizes: "192x192", type: "image/png" }]
        };
        const manifestBlob = new Blob([JSON.stringify(manifest)], {type: 'application/json'});
        const manifestUrl = URL.createObjectURL(manifestBlob);
        const link = document.createElement('link');
        link.rel = 'manifest';
        link.href = manifestUrl;
        document.head.appendChild(link);
    </script>
